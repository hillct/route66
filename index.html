Route66

NPM Github Twitter

Route66 was created to replace good old and removed connect.router middleware for Connect. It offers simple route declaration, in-place middleware support and declaring params in requests.

Installation

npm install route66

Usage

connect = require 'connect'
router = require 'route66'
app = do connect

router.get '/', (req, res) -> # just a simple route
	do res.end

router.get '/new', (req, res, next) -> # route with middleware
	do next
, (req, res) ->
	do res.end

router.post '/', (req, res) -> # POST, this time
	do res.end

router.get '/:id', (req, res) -> # request with parameters
	# req.params.id
	do res.end

app.use router
app.listen 3000
